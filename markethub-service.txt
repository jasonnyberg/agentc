# Copy to system location
sudo cp markethub/markethub-backend.service /etc/systemd/system/
# Reload systemd, enable and start
sudo systemctl daemon-reload
sudo systemctl enable markethub-backend
sudo systemctl start markethub-backend
# Check status
sudo systemctl status markethub-backend
# Stop / restart
sudo systemctl stop markethub-backend
sudo systemctl restart markethub-backend
# Switch to TradeStation without editing the file:
sudo systemctl edit markethub-backend
